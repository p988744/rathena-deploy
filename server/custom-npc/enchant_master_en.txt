		mes "^FF0000This module has already reached its maximum count on this equipment.^000000";
		mes "Please choose a different module.";
		close;
	}

	mes "[Enchant Master]";
	mes "Apply ^0000CD" + getitemname(.@module_id) + "^000000 to ^FF0000" + .@equip_name$ + "^000000?";
	mes "^808080This cannot be undone.^000000";
	next;
	if (select( "Yes, enchant it!", "Cancel" ) == 2) {
		mes "[Enchant Master]";
		mes "Cancelled.";

		end;

	for (.@i = 3; .@i > 0; .@i--) {
		if (.@card[.@i] == 0) {
			.@card[.@i] = .@enchant_id;
			break;
		}
	}
	delequip .@part;
	getitem2 .@equip_id, 1, 1, .@refine, 0, .@card[0], .@card[1], .@card[2], .@card[3];

	mes "[Enchant Master]";
	mes "Success! ^0000CD" + getitemname(.@module_id) + "^000000 has been applied.";
	mes "Make the most of your Automatic equipment!";
