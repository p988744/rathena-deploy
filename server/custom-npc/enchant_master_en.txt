	    callfunc("F_IsEquipRefineHack", .@part, .@equip_refine) ||
	    callfunc("F_IsEquipCardHack", .@part, .@card[0], .@card[1], .@card[2], .@card[3]))
		end;
	delequip .@part;
	.@card[0] = .@enchant;
	mes "[Enchant Master]";
	mes "^00CC00Success!^000000 ^0000CD" + getitemname(.@enchant) + "^000000 applied!";
	getitem2 .@equip_id, 1, 1, .@equip_refine, 0, .@card[0], .@card[1], .@card[2], .@card[3];
	close;

	return;
}
